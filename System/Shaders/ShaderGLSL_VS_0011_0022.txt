layout (location = 5) in vec2 DetailTexCoords;

{
    mat4 modelviewprojMat;
};

out vec2 vDetailTexCoords;

{

	vDetailTexCoords	= DetailTexCoords;

	gl_Position = modelviewprojMat * vec4(v_coord, 1.0);
}
