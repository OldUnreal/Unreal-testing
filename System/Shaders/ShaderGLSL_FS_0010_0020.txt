
//== SHADER COMPILE STATUS = TRUE
#version 330
uniform sampler2D Texture0;
uniform vec4 drawcolor;
uniform float AlphaThreshold;

in vec2 vFogMapCoords;

out vec4 FragColor; 

void main (void)  
{  
	vec4 Color = texture(Texture0, vFogMapCoords); // For FogMap
	if(Color.a <= AlphaThreshold) 
        discard;

	vec4 TotalColor  = Color*drawcolor; // Add drawcolor.
	
	FragColor = TotalColor;             
}
